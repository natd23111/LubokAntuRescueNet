rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Temporarily allow seeding - REMOVE AFTER SEEDING IS COMPLETE
    function canSeed() {
      return request.auth != null;
    }

    // Helper functions
    function isSignedIn() {
      return request.auth != null;
    }
    
    function isAdmin() {
      return isSignedIn() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    function isValidImageFields() {
      return (
        // if image_urls present it must be a list with at most 3 items
        (!('image_urls' in request.resource.data) ||
        (request.resource.data.image_urls is list && request.resource.data.image_urls.size() <= 3))
        &&
        // if image_url present it must be a string
        (!('image_url' in request.resource.data) ||
        (request.resource.data.image_url is string))
      );
    }
    
    // Users can read/write their own profile OR for seeding
    match /users/{userId} {
      allow read: if request.auth.uid == userId;
      allow write: if request.auth.uid == userId || canSeed();
    }
    
    // Aid programs - public read, admin write or seeding
    match /aid_programs/{document=**} {
      allow read: if true;
      allow write: if isAdmin() || canSeed();
    }
    
    // Emergency reports - anyone logged in can create, only owner/admin can update/delete
    match /emergency_reports/{reportId} {
      allow read: if true;
      allow create: if isSignedIn()
                    && request.resource.data.user_id == request.auth.uid
                    && isValidImageFields();
      allow update: if isSignedIn()
                    && (resource.data.user_id == request.auth.uid || isAdmin())
                    && isValidImageFields();
      allow delete: if isAdmin();
    }
    
    // Aid requests - users can create/read their own, admins can read/update/delete all
    match /aid_requests/{requestId} {
      allow read: if isSignedIn()
                  && (resource.data.user_id == request.auth.uid || isAdmin());
      allow create: if isSignedIn()
                    && request.resource.data.user_id == request.auth.uid;
      allow update: if isSignedIn()
                    && (resource.data.user_id == request.auth.uid || isAdmin());
      allow delete: if isAdmin();
    }
    
    // Metadata for counters - authenticated users or seeding
    match /_metadata/{document=**} {
      allow read, write: if isSignedIn() || canSeed();
    }
    
    // Test data - open for testing
    match /test_data/{document=**} {
      allow read, write: if true;
    }
  }
}
